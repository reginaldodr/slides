\subsection{Substituição Trigonométrica}


\begin{frame}
\frametitle{Substituição Trigonométrica }
\begin{scriptsize}

\uncover<1->{\begin{block}{ }
As substituições trigonométricas são úteis para calcular integrais que envolvem 
$$\sqrt{a^2-x^2},\ \sqrt{a^2+x^2},\ \mbox{ e } \sqrt{x^2-a^2}$$ 
\end{block}}


\uncover<2->{\begin{exe}
\end{frame}

\begin{frame}
\frametitle{}
\begin{scriptsize}

\uncover<1->{\begin{block}{ } A substituição 
$$z=\tg\frac{x}{2}$$
reduz o problema de integrar uma expressão racional de $\sen x$ e $\cos x$ ao problema de integrar uma função racional de z.
\end{block}\end{exe}}

\uncover<2->{\begin{exe}
\begin{enumerate}
\item $\dps \int \frac{1}{1+\cos x}dx$
\item $\dps \int\frac{1}{2+\sen x}dx$
\end{enumerate}\end{exe}}

\end{scriptsize}
\end{frame}

\begin{frame}
\begin{small}
\begin{exer}
Calcule as integrais
\begin{enumerate}
\item  $\dps\int \sqrt{1-9t^2} dt$.

\item  $\dps \int \frac{1}{\sen t-\cos t}$.
\end{enumerate}

\end{exer}

\end{small}


\end{frame}
